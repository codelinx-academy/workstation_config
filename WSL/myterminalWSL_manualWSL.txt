1. sudo apt update && sudo apt upgrade -y
2. sudo apt install git curl -y
3. sudo apt install gnome-tweaks -y
4. sudo apt install zsh -y
5. sudo chsh -s /bin/zsh
6. sudo sh -c "$(curl -fsSL https://raw.github.com/ohmyzsh/ohmyzsh/master/tools/install.sh)"
7. sudo git clone https://github.com/zsh-users/zsh-syntax-highlighting.git ${ZSH_CUSTOM:-~/.oh-my-zsh/custom}/plugins/zsh-syntax-highlighting
8. sudo mkdir ~/.fonts
9. sudo curl -L https://github.com/romkatv/powerlevel10k-media/raw/master/MesloLGS%20NF%20Regular.ttf -o ~/.fonts/MesloLGS\ NF\ Regular.ttf
10. gsettings set org.gnome.desktop.interface monospace-font-name "MesloLGS NF Regular 11"
11. sudo git clone --depth=1 https://github.com/romkatv/powerlevel10k.git ${ZSH_CUSTOM:-$HOME/.oh-my-zsh/custom}/themes/powerlevel10k
12. sed -i 's#^ZSH_THEME=.*#ZSH_THEME="powerlevel10k/powerlevel10k"#' ~/.zshrc && sed -i 's#^plugins=.*#plugins=(git zsh-syntax-highlighting)#' ~/.zshrc
