export VAGRANT_WSL_ENABLE_WINDOWS_ACCESS="1"
export PATH="$PATH:/mnt/c/Program\ Files/Oracle/VirtualBox"

03. lincar usando

04. aplicar lab:
Vagrant.configure("2") do |config|
  # Função para definir uma VM.
  def define_vm(config, name, box_manager)
    config.vm.define name do |vm| # Define a VM com um nome específico.
      vm.vm.box = box_manager[0] # Define a imagem da VM.
    
      vm.vm.provider "virtualbox" do |vb| # Configurações específicas do VirtualBox.
        vb.name = name # Nome da VM no VirtualBox.
      end
      
      # Comandos iniciais para a VM.
      vm.vm.provision 'shell', inline: <<-SHELL
        sudo #{box_manager[1]} -y #{box_manager[1] == 'zypper' ? 'refresh' : 'update'} # Atualiza os pacotes.
        sudo #{box_manager[1]} -y install openssh-server # Instala o servidor SSH.
      SHELL
    end
  end

  # Hash com as boxes e gerenciadores de pacotes para diferentes distribuições Linux.
  distro = {ubuntu_box:['ubuntu/trusty64', 'apt-get'],
    rhel_box:['generic/rhel7', 'yum'],
    opensuse_box:['generic/opensuse42', 'zypper']}

  # Define seis VMs com diferentes distribuições Linux.
  define_vm(config, 'lnx01', distro[:ubuntu_box])
  define_vm(config, 'lnx02', distro[:ubuntu_box])
  define_vm(config, 'lnx03', distro[:rhel_box])
  define_vm(config, 'lnx04', distro[:rhel_box])
  define_vm(config, 'lnx05', distro[:opensuse_box])
  define_vm(config, 'lnx06', distro[:opensuse_box])
end
